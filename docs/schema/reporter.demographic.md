```sql
CREATE VIEW demographic AS (
 SELECT u.id,
    u.dob,
        CASE
            WHEN (u.dob IS NULL) THEN 'Adult'::text
            WHEN (age((u.dob)::timestamp with time zone) > '18 years'::interval) THEN 'Adult'::text
            ELSE 'Juvenile'::text
        END AS general_division
   FROM actor.usr u
)
```
