CREATE VIEW fund_combined_balance AS (
 SELECT c.fund,
    (c.amount - COALESCE(d.amount, 0.0)) AS amount
   FROM (acq.fund_allocation_total c
     LEFT JOIN acq.fund_debit_total d USING (fund))
)
