<summary><strong>Table Definition</strong></summary>

```sql
CREATE VIEW xact_paid_totals AS (
 SELECT b.xact,
    sum(
        CASE
            WHEN b.voided THEN (0)::numeric
            ELSE b.amount
        END) AS unvoided,
    sum(
        CASE
            WHEN b.voided THEN b.amount
            ELSE (0)::numeric
        END) AS voided,
    sum(b.amount) AS total
   FROM money.payment b
  GROUP BY b.xact
)
```
